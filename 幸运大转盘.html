<script>

    (function() {

        // ---------- 转盘配置 ----------

        const COLORS = [

            "#E34234", // 红 炽热红

            "#FF8C42", // 橙 暖橙

            "#F3D572", // 黄 灿金

            "#5A9E4E", // 绿 翠林

            "#2F9F9F", // 青 深海青

            "#4A7B9D", // 蓝 天湖蓝

            "#9B59B6"  // 紫 罗兰紫

        ];

        const COLOR_NAMES = ["红", "橙", "黄", "绿", "青", "蓝", "紫"];

        const SECTOR_COUNT = 7;

        const SECTOR_ANGLE = (Math.PI * 2) / SECTOR_COUNT; // 每个扇形弧度

       

        // 固定指针方向 (向上: -PI/2 弧度)

        const POINTER_ANGLE = -Math.PI / 2;

       

        // canvas 元素与上下文

        const canvas = document.getElementById('wheelCanvas');

        const ctx = canvas.getContext('2d');

        let width, height, centerX, centerY, radius;

       

        // 转盘当前绝对旋转弧度 (扇区偏移量)

        let currentRotation = 0;     // 0 ~ 2PI

       

        // 动画相关

        let animating = false;

        let animFrameId = null;

        let startRotation = 0;

        let totalDelta = 0;           // 总共需要旋转的增量弧度 (正值)

        let animStartTime = 0;

        const ANIM_DURATION = 1800;    // 毫秒

       

        // 抽奖状态 (每日限制)

        let todayHasSpun = false;      // 今日是否已经转动过

        let todayLuckyColorIndex = null;   // 今日幸运色索引 0-6

        let currentDateStr = "";

       

        // DOM 元素

        const spinBtn = document.getElementById('spinButton');

        const resultMsgDiv = document.getElementById('resultMessage');

        const luckyColorDot = document.getElementById('luckyColorDot');

        const luckyColorNameSpan = document.getElementById('luckyColorName');

        const dailyStatusTip = document.getElementById('dailyStatusTip');

       

        // ---------- 辅助函数 ----------

        function getTodayDateStr() {

            const today = new Date();

            return `${today.getFullYear()}-${today.getMonth()+1}-${today.getDate()}`;

        }

       

        // 初始化/读取存储: 今日幸运色、今日是否已抽奖、当前转盘角度 (角度保持体验)

        function loadDailyData() {

            const today = getTodayDateStr();

            currentDateStr = today;

           

            // 读取幸运色 (按日期存储)

            const storedLucky = localStorage.getItem(`luckyColor_${today}`);

            if (storedLucky !== null && !isNaN(parseInt(storedLucky))) {

                todayLuckyColorIndex = parseInt(storedLucky);

                if (todayLuckyColorIndex < 0 || todayLuckyColorIndex >= SECTOR_COUNT) {

                    // 容错

                    todayLuckyColorIndex = Math.floor(Math.random() * SECTOR_COUNT);

                }

            } else {

                // 随机生成今日幸运色

                todayLuckyColorIndex = Math.floor(Math.random() * SECTOR_COUNT);

                localStorage.setItem(`luckyColor_${today}`, todayLuckyColorIndex);

            }

           

            // 读取今日是否已经抽过奖

            const spinDone = localStorage.getItem(`spinDone_${today}`);

            todayHasSpun = (spinDone === 'true');

           

            // 读取上次保存的转盘角度 (全局存储，与日期无关，提升连续体验)

            const savedRotation = localStorage.getItem(`wheelRotation`);

            if (savedRotation !== null && !isNaN(parseFloat(savedRotation))) {

                let rot = parseFloat(savedRotation) % (Math.PI * 2);

                if (rot < 0) rot += Math.PI * 2;

                currentRotation = rot;

            } else {

                currentRotation = 0;

            }

           

            // 更新UI幸运色展示

            updateLuckyDisplay();

            // 根据今日是否抽过奖，更新按钮与提示

            updateButtonAndMessageByStatus();

        }

       

        // 更新幸运色UI

        function updateLuckyDisplay() {

            if (todayLuckyColorIndex !== null) {

                const luckyColorHex = COLORS[todayLuckyColorIndex];

                const luckyName = COLOR_NAMES[todayLuckyColorIndex];

                luckyColorDot.style.backgroundColor = luckyColorHex;

                luckyColorNameSpan.innerText = luckyName;

                luckyColorNameSpan.style.color = luckyColorHex;

                luckyColorNameSpan.style.textShadow = "0 0 4px black";

            }

        }

       

        // 根据今日是否已抽奖更新按钮和消息区

        function updateButtonAndMessageByStatus() {

            if (todayHasSpun) {

                spinBtn.disabled = true;

                dailyStatusTip.innerHTML = "⏰ 今日运势已揭晓 · 明日再来 ⏰";

                // 不覆盖结果消息，但如果结果消息还是欢迎语，可改为温和提醒

                if (resultMsgDiv.innerText.includes("点击转盘") || resultMsgDiv.innerText.includes("🌈")) {

                    resultMsgDiv.innerHTML = "✨ 今日你已经转动过轮盘啦！ 明天再来试试手气 ✨";

                }

            } else {

                spinBtn.disabled = false;

                dailyStatusTip.innerHTML = "🍀 今日免费机会 · 点击转盘抽取幸运色 🍀";

            }

        }

       

        // 保存转盘角度 (每次动画结束/页面关闭前保存)

        function saveCurrentRotation() {

            localStorage.setItem(`wheelRotation`, currentRotation);

        }

       

        // 绘制转盘 (根据currentRotation弧度)

        function drawWheel() {

            if (!ctx) return;

            ctx.clearRect(0, 0, width, height);

           

            // 绘制底色背景透明圆环质感

            ctx.shadowBlur = 0;

            for (let i = 0; i < SECTOR_COUNT; i++) {

                const start = i * SECTOR_ANGLE + currentRotation;

                const end = (i + 1) * SECTOR_ANGLE + currentRotation;

               

                // 绘制扇形

                ctx.beginPath();

                ctx.moveTo(centerX, centerY);

                ctx.arc(centerX, centerY, radius, start, end);

                ctx.closePath();

                ctx.fillStyle = COLORS[i];

                ctx.fill();

                // 描边白边分割

                ctx.save();

                ctx.shadowBlur = 0;

                ctx.strokeStyle = "#FFF9E8";

                ctx.lineWidth = 2.5;

                ctx.beginPath();

                ctx.moveTo(centerX, centerY);

                ctx.arc(centerX, centerY, radius, start, end);

                ctx.lineTo(centerX, centerY);

                ctx.stroke();

                ctx.restore();

               

                // 绘制文字 (扇形中间)

                const midAngle = start + SECTOR_ANGLE / 2;

                const textRadius = radius * 0.68;

                const x = centerX + Math.cos(midAngle) * textRadius;

                const y = centerY + Math.sin(midAngle) * textRadius;

                ctx.font = `bold ${Math.floor(radius * 0.12)}px "Segoe UI", "PingFang SC", "Microsoft YaHei"`;

                ctx.fillStyle = "#1F1B14";

                ctx.shadowBlur = 0;

                ctx.shadowColor = "rgba(0,0,0,0)";

                ctx.textAlign = "center";

                ctx.textBaseline = "middle";

                // 添加字体白边效果更清晰

                ctx.fillStyle = "#FFF7E8";

                ctx.shadowBlur = 3;

                ctx.shadowColor = "rgba(0,0,0,0.4)";

                ctx.fillText(COLOR_NAMES[i], x, y);

                ctx.fillStyle = "#2D2418";

                ctx.shadowBlur = 1;

                ctx.fillText(COLOR_NAMES[i], x-0.5, y-0.5);

                ctx.fillStyle = "#FFF2D7";

                ctx.fillText(COLOR_NAMES[i], x, y);

            }

           

            // 绘制中心装饰圆

            ctx.beginPath();

            ctx.arc(centerX, centerY, radius * 0.12, 0, Math.PI * 2);

            ctx.fillStyle = "#FAD460";

            ctx.shadowBlur = 8;

            ctx.fill();

            ctx.beginPath();

            ctx.arc(centerX, centerY, radius * 0.07, 0, Math.PI * 2);

            ctx.fillStyle = "#FFE3A3";

            ctx.fill();

            ctx.shadowBlur = 0;

        }

       

        // 根据当前旋转角度，获取指针指向的扇区索引 (基于POINTER_ANGLE)

        function getCurrentSectorIndex() {

            // 指针固定方向pointerAngle，转盘旋转角度currentRotation，扇区0范围 [0+rot, sectorAngle+rot)

            // 计算指针相对于转盘实际落在哪个区域: 指针角度(绝对) = POINTER_ANGLE

            // 偏移后的扇区边界：每个扇区原始范围 i*sectorAngle ~ (i+1)*sectorAngle 再整体偏移 currentRotation

            // 所以需要求解 i 使得: i*sectorAngle + currentRotation <= pointerAngle < (i+1)*sectorAngle + currentRotation

            let rawAngle = POINTER_ANGLE;

            // 将指针角度规范到 [0, 2PI)

            let pointerNorm = rawAngle;

            if (pointerNorm < 0) pointerNorm += Math.PI * 2;

           

            for (let i = 0; i < SECTOR_COUNT; i++) {

                let start = i * SECTOR_ANGLE + currentRotation;

                let end = (i+1) * SECTOR_ANGLE + currentRotation;

                // 规范化到 [0, 2PI) 比较（避免边界问题）

                let startNorm = start % (Math.PI * 2);

                let endNorm = end % (Math.PI * 2);

                if (startNorm < 0) startNorm += Math.PI * 2;

                if (endNorm < 0) endNorm += Math.PI * 2;

               

                if (startNorm < endNorm) {

                    if (pointerNorm >= startNorm && pointerNorm < endNorm) {

                        return i;

                    }

                } else {

                    // 跨越 2PI 区间

                    if (pointerNorm >= startNorm || pointerNorm < endNorm) {

                        return i;

                    }

                }

            }

            // fallback 容错

            return 0;

        }

       

        // 根据目标扇区索引，计算期望的最终旋转角度 (让指针指向该扇区的中心区域)

        function getTargetRotationForSector(targetIdx) {

            // 我们希望指针方向正好落在目标扇区的中间位置

            const targetSectorMid = targetIdx * SECTOR_ANGLE + SECTOR_ANGLE / 2;

            // 指针方向固定，转盘需要旋转到 rotationTarget 使得：

            // (targetSectorMid + rotationTarget) % 2PI = POINTER_ANGLE (mod 2PI)

            // 即 rotationTarget = POINTER_ANGLE - targetSectorMid (mod 2PI)

            let rawTarget = POINTER_ANGLE - targetSectorMid;

            let modTarget = ((rawTarget % (Math.PI * 2)) + Math.PI * 2) % (Math.PI * 2);

            return modTarget;

        }

       

        // 动画: 从 currentRotation 旋转 totalDeltaRad 弧度 (总增加弧度)

        function startAnimation(finalTargetRotation) {

            if (animating) return;

            // 计算最短增量 (正向旋转增加totalDelta，最终到达finalTargetRotation)

            let start = currentRotation;

            let target = finalTargetRotation;

            // 标准化 target 到 [0, 2PI)

            target = ((target % (2 * Math.PI)) + 2 * Math.PI) % (2 * Math.PI);

           

            // 计算正向增量 delta_needed (从start到target的最小正向弧长)

            let deltaNeeded = target - start;

            deltaNeeded = ((deltaNeeded % (2 * Math.PI)) + 2 * Math.PI) % (2 * Math.PI);

           

            // 增加随机整圈数 5~10 圈，使转动更过瘾

            const extraRounds = 5 + Math.floor(Math.random() * 6);  // 5~10圈

            const extraAngle = extraRounds * 2 * Math.PI;

            totalDelta = deltaNeeded + extraAngle;

           

            startRotation = currentRotation;

            animStartTime = performance.now();

            animating = true;

           

            function step(now) {

                const elapsed = now - animStartTime;

                let t = Math.min(1, elapsed / ANIM_DURATION);

                // 缓动 cubic out

                const easeOut = 1 - Math.pow(1 - t, 3);

                const newRot = startRotation + totalDelta * easeOut;

                currentRotation = newRot % (2 * Math.PI);

                drawWheel();

               

                if (t < 1) {

                    animFrameId = requestAnimationFrame(step);

                } else {

                    // 动画结束，精准对齐最终目标角度 (避免浮点误差)

                    currentRotation = finalTargetRotation % (2 * Math.PI);

                    if (currentRotation < 0) currentRotation += 2 * Math.PI;

                    drawWheel();

                    finishSpinAndCheckResult();

                    cancelAnimationFrame(animFrameId);

                    animating = false;

                    animFrameId = null;

                }

            }

           

            if (animFrameId) cancelAnimationFrame(animFrameId);

            animFrameId = requestAnimationFrame(step);

        }

       

        // 抽奖完成后的处理: 判定结果，存储状态，更新界面

        function finishSpinAndCheckResult() {

            if (!todayHasSpun) {

                // 抽奖完成，获取当前指针指向的扇区索引

                const landedIndex = getCurrentSectorIndex();

                const landedColorName = COLOR_NAMES[landedIndex];

                const isWin = (landedIndex === todayLuckyColorIndex);

               

                // 存储今日已抽奖标志

                todayHasSpun = true;

                localStorage.setItem(`spinDone_${currentDateStr}`, 'true');

               

                // 保存当前转盘角度 (抽完后)

                saveCurrentRotation();

               

                // 更新UI提示

                const luckyColor = COLOR_NAMES[todayLuckyColorIndex];

                if (isWin) {

                    resultMsgDiv.innerHTML = `🎉🎉 鸿运当头！ 指针停在【${landedColorName}】！ 正是今日幸运色“${luckyColor}” 🎉🎉<br>✨ 恭喜选中幸运色！好运相随 ✨`;

                    resultMsgDiv.style.background = "#d4af37aa";

                    resultMsgDiv.style.borderLeftColor = "#ffdd88";

                } else {

                    resultMsgDiv.innerHTML = `🍃 转盘停在【${landedColorName}】... 今日幸运色是“${luckyColor}”  🍃<br>⭐ 明日再接再厉，再试手气 ⭐`;

                    resultMsgDiv.style.background = "#1f2a36aa";

                    resultMsgDiv.style.borderLeftColor = "#bc9a5a";

                }

                // 禁用按钮 并 更新每日提示

                spinBtn.disabled = true;

                dailyStatusTip.innerHTML = "⏰ 今日运势已揭晓 · 明日再来 ⏰";

            } else {

                // 理论上动画触发前已经拦截，但双重保险

                resultMsgDiv.innerHTML = "⚠️ 今日已经转动过了，明天再试试吧～";

                spinBtn.disabled = true;

            }

        }

       

        // 核心: 点击抽奖

        function onSpin() {

            if (animating) {

                resultMsgDiv.innerHTML = "🎡 转盘旋转中，请稍后...";

                return;

            }

            if (todayHasSpun) {

                resultMsgDiv.innerHTML = "🌙 今日机会已用尽，明日再来转动轮盘，邂逅幸运色彩吧 🌙";

                return;

            }

           

            // 随机抽取目标扇区索引 (0-6完全随机)

            const targetSector = Math.floor(Math.random() * SECTOR_COUNT);

            // 获取要让指针指向该扇区所需要的最终旋转弧度

            const targetRotation = getTargetRotationForSector(targetSector);

           

            // 开始动画，目标旋转值targetRotation

            startAnimation(targetRotation);

           

            // 额外在动画开始时给个小提示（取消旧结果样式）

            resultMsgDiv.innerHTML = "🌀 命运轮盘转动中... 幸运即将揭晓 🌀";

            resultMsgDiv.style.background = "#0f0e17aa";

            resultMsgDiv.style.borderLeftColor = "gold";

        }

       

        // 重置画布尺寸响应式，并重绘

        function resizeAndDraw() {

            const container = canvas.parentElement;

            const maxSize = Math.min(container.clientWidth, 500);

            canvas.style.width = `${maxSize}px`;

            canvas.style.height = `${maxSize}px`;

            // 实际画布分辨率保持500x500防止模糊

            canvas.width = 500;

            canvas.height = 500;

            width = 500;

            height = 500;

            centerX = width / 2;

            centerY = height / 2;

            radius = width * 0.45;

            drawWheel();

        }

       

        // 监听存储跨日更新 (页面可见时检查日期是否变更，刷新状态)

        function checkDateChangeAndReset() {

            const newDate = getTodayDateStr();

            if (newDate !== currentDateStr) {

                // 日期变了，重新加载所有数据 (幸运色，抽奖状态)

                loadDailyData();

                // 重绘转盘

                drawWheel();

                // 如果之前的结果消息还保留，重置为默认消息

                if (!todayHasSpun) {

                    resultMsgDiv.innerHTML = "🌈 新的一天！转动转盘，争取选中今日幸运色 🌈";

                    resultMsgDiv.style.background = "#0f0e17aa";

                } else {

                    resultMsgDiv.innerHTML = "✨ 今日已经抽过幸运色啦，明天再玩 ✨";

                }

                // 按钮状态已在 loadDailyData 中通过 updateButtonAndMessageByStatus 更新

            }

        }

       

        // 页面可见性变化时跨天刷新

        function handleVisibilityChange() {

            if (!document.hidden) {

                checkDateChangeAndReset();

            }

        }

       

        // 保存角度到本地，页面关闭前保存一次

        window.addEventListener('beforeunload', () => {

            if (!animating) {

                saveCurrentRotation();

            }

        });

       

        // 初始化一切

        function init() {

            resizeAndDraw();

            loadDailyData();    // 会设置 todayLuckyColorIndex, todayHasSpun, currentRotation

            drawWheel();        // 根据currentRotation绘制

           

            spinBtn.addEventListener('click', onSpin);

            window.addEventListener('resize', () => {

                resizeAndDraw();

            });

            document.addEventListener('visibilitychange', handleVisibilityChange);

            // 每60秒检查一次跨日 (可选)

            setInterval(() => {

                checkDateChangeAndReset();

            }, 60000);

        }

       

        init();

    })();

</script>
